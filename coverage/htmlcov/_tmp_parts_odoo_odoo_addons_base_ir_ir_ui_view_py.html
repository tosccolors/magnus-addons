37def keep_query(*keep_params, **additional_params): 

38 """ 

39 Generate a query string keeping the current request querystring's parameters specified 

40 in ``keep_params`` and also adds the parameters specified in ``additional_params``. 

41 

42 Multiple values query string params will be merged into a single one with comma seperated 

43 values. 

44 

45 The ``keep_params`` arguments can use wildcards too, eg: 

46 

47 keep_query('search', 'shop_*', page=4) 

48 """ 

49 if not keep_params and not additional_params: 

50 keep_params = ('*',) 

51 params = additional_params.copy() 

52 qs_keys = request.httprequest.args.keys() if request else [] 

53 for keep_param in keep_params: 

54 for param in fnmatch.filter(qs_keys, keep_param): 

55 if param not in additional_params and param in qs_keys: 

56 params[param] = request.httprequest.args.getlist(param) 

57 return werkzeug.urls.url_encode(params)

95def get_view_arch_from_file(filename, xmlid): 

96 doc = etree.parse(filename) 

97 node = None 

98 for n in doc.xpath('//*[@id="%s"]' % (xmlid)): 

99 if n.tag in ('template', 'record'): 

100 node = n 

101 break 

102 if node is None: 

103 # fallback search on template with implicit module name 

104 for n in doc.xpath('//*[@id="%s"]' % (xmlid.split('.')[1])): 

105 if n.tag in ('template', 'record'): 

106 node = n 

107 break 

108 if node is not None: 

109 if node.tag == 'record': 

110 field = node.find('field[@name="arch"]') 

111 _fix_multiple_roots(field) 

112 inner = ''.join([etree.tostring(child) for child in field.iterchildren()]) 

113 return field.text + inner 

114 elif node.tag == 'template': 

115 # The following dom operations has been copied from convert.py's _tag_template() 

116 if not node.get('inherit_id'): 

117 node.set('t-name', xmlid) 

118 node.tag = 't' 

119 else: 

120 node.tag = 'data' 

121 node.attrib.pop('id', None) 

122 return etree.tostring(node) 

123 _logger.warning("Could not find view arch definition in file '%s' for xmlid '%s'", filename, xmlid) 

124 return None

258 def _valid_inheritance(self, arch): 

259 """ Check whether view inheritance is based on translated attribute. """

479 def locate_node(self, arch, spec): 

480 """ Locate a node in a source (parent) architecture. 

481 

482 Given a complete source (parent) architecture (i.e. the field 

483 `arch` in a view), and a 'spec' node (a node in an inheriting 

484 view that specifies the location in the source view of what 

485 should be changed), return (if it exists) the node in the 

486 source view matching the specification. 

487 

488 :param arch: a parent architecture to modify 

489 :param spec: a modifying node in an inheriting view 

490 :return: a node in the source matching the spec 

491 """ 

492 if spec.tag == 'xpath': 

493 nodes = arch.xpath(spec.get('expr')) 

494 return nodes[0] if nodes else None 

495 elif spec.tag == 'field': 

496 # Only compare the field name: a field can be only once in a given view 

497 # at a given level (and for multilevel expressions, we should use xpath 

498 # inheritance spec anyway). 

499 for node in arch.iter('field'): 

500 if node.get('name') == spec.get('name'): 

501 return node 

502 return None 

503 

504 for node in arch.iter(spec.tag): 

505 if isinstance(node, SKIPPED_ELEMENT_TYPES): 

506 continue 

507 if all(node.get(attr) == spec.get(attr) for attr in spec.attrib 

508 if attr not in ('position', 'version')): 

509 # Version spec should match parent's root element's version 

510 if spec.get('version') and spec.get('version') != arch.get('version'): 

511 return None 

512 return node 

513 return None

533 architecture) given by an inheriting view. 

534 

535 :param Element source: a parent architecture to modify 

536 :param Elepect specs_tree: a modifying architecture in an inheriting view 

537 :param inherit_id: the database id of specs_arch 

538 :return: a modified source where the specs are applied 

539 :rtype: Element 

540 """ 

541 # Queue of specification nodes (i.e. nodes describing where and 

542 # changes to apply to some parent architecture). 

543 specs = [specs_tree] 

544 

545 while len(specs): 

546 spec = specs.pop(0) 

547 if isinstance(spec, SKIPPED_ELEMENT_TYPES): 

548 continue 

549 if spec.tag == 'data': 

550 specs += [c for c in spec] 

551 continue 

552 node = self.locate_node(source, spec) 

553 if node is not None: 

554 pos = spec.get('position', 'inside') 

555 if pos == 'replace': 

556 for loc in spec.xpath(".//*[text()='$0']"): 

557 loc.text = '' 

558 loc.append(copy.deepcopy(node)) 

559 if node.getparent() is None: 

560 source = copy.deepcopy(spec[0]) 

561 else: 

562 for child in spec: 

563 node.addprevious(child) 

564 node.getparent().remove(node) 

565 elif pos == 'attributes': 

566 for child in spec.getiterator('attribute'): 

567 attribute = child.get('name') 

568 value = child.text or '' 

569 if child.get('add') or child.get('remove'): 

570 assert not child.text 

571 separator = child.get('separator', ',') 

572 if separator == ' ': 

573 separator = None # squash spaces 

574 to_add = filter(bool, map(str.strip, child.get('add', '').split(separator))) 

575 to_remove = map(str.strip, child.get('remove', '').split(separator)) 

576 values = map(str.strip, node.get(attribute, '').split(separator)) 

577 value = (separator or ' ').join(filter(lambda s: s not in to_remove, values) + to_add) 

578 if value: 

579 node.set(attribute, value) 

580 elif attribute in node.attrib: 

581 del node.attrib[attribute] 

582 else: 

583 sib = node.getnext() 

584 for child in spec: 

585 if pos == 'inside': 

586 node.append(child) 

587 elif pos == 'after': 

588 if sib is None: 

589 node.addnext(child) 

590 node = child 

591 else: 

592 sib.addprevious(child) 

593 elif pos == 'before': 

594 node.addprevious(child) 

595 else: 

596 self.raise_view_error(_("Invalid position attribute: '%s'") % pos, inherit_id) 

597 else: 

598 attrs = ''.join([ 

599 ' %s="%s"' % (attr, spec.get(attr)) 

600 for attr in spec.attrib 

601 if attr != 'position' 

602 ]) 

603 tag = "<%s%s>" % (spec.tag, attrs) 

604 self.raise_view_error(_("Element '%s' cannot be located in parent view") % tag, inherit_id) 

605 

606 return source

608 @api.model 

609 def apply_view_inheritance(self, source, source_id, model, root_id=None): 

610 """ Apply all the (directly and indirectly) inheriting views. 

611 

612 :param source: a parent architecture to modify (with parent modifications already applied) 

613 :param source_id: the database view_id of the parent view 

614 :param model: the original model for which we create a view (not 

615 necessarily the same as the source's model); only the inheriting 

616 views with that specific model will be applied. 

617 :return: a modified source where all the modifying architecture are applied 

618 """ 

619 if root_id is None: 

620 root_id = source_id 

621 sql_inherit = self.get_inheriting_views_arch(source_id, model) 

622 for (specs, view_id) in sql_inherit: 

623 specs_tree = etree.fromstring(specs.encode('utf-8')) 

624 if self._context.get('inherit_branding'): 

625 self.inherit_branding(specs_tree, view_id, root_id) 

626 source = self.apply_inheritance_specs(source, specs_tree, view_id) 

627 source = self.apply_view_inheritance(source, view_id, model, root_id=root_id) 

628 return source

631 def read_combined(self, fields=None): 

632 """ 

633 Utility function to get a view combined with its inherited views. 

634 

635 * Gets the top of the view tree if a sub-view is requested 

636 * Applies all inherited archs on the root view 

637 * Returns the view with all requested fields 

638 .. note:: ``arch`` is always added to the fields list even if not 

639 requested (similar to ``id``) 

640 """ 

641 # introduce check_view_ids in context 

642 if 'check_view_ids' not in self._context: 

643 self = self.with_context(check_view_ids=[]) 

644 

645 check_view_ids = self._context['check_view_ids'] 

646 

647 # if view_id is not a root view, climb back to the top. 

648 root = self 

649 while root.mode != 'primary': 

650 # Add inherited views to the list of loading forced views 

651 # Otherwise, inherited views could not find elements created in their direct parents if that parent is defined in the same module 

652 check_view_ids.append(root.id) 

653 root = root.inherit_id 

654 

655 # arch and model fields are always returned 

656 if fields: 

657 fields = list({'arch', 'model'}.union(fields)) 

658 

659 # read the view arch 

660 [view_data] = root.read(fields=fields) 

661 view_arch = etree.fromstring(view_data['arch'].encode('utf-8')) 

662 if not root.inherit_id: 

663 arch_tree = view_arch 

664 else: 

665 parent_view = root.inherit_id.read_combined(fields=fields) 

666 arch_tree = etree.fromstring(parent_view['arch']) 

667 arch_tree = self.apply_inheritance_specs(arch_tree, view_arch, parent_view['id']) 

668 

669 if self._context.get('inherit_branding'): 

670 arch_tree.attrib.update({ 

671 'data-oe-model': 'ir.ui.view', 

672 'data-oe-id': str(root.id), 

673 'data-oe-field': 'arch', 

674 }) 

675 

676 # and apply inheritance 

677 arch = self.apply_view_inheritance(arch_tree, root.id, self.model) 

678 

679 return dict(view_data, arch=etree.tostring(arch, encoding='utf-8'))

681 def _apply_group(self, model, node, modifiers, fields): 

682 """Apply group restrictions, may be set at view level or model level:: 

683 * at view level this means the element should be made invisible to 

684 people who are not members 

685 * at model level (exclusively for fields, obviously), this means 

686 the field should be completely removed from the view, as it is 

687 completely unavailable for non-members 

688 

689 :return: True if field should be included in the result of fields_view_get 

690 """ 

691 Model = self.env[model] 

692 

693 if node.tag == 'field' and node.get('name') in Model._fields: 

694 field = Model._fields[node.get('name')] 

695 if field.groups and not self.user_has_groups(groups=field.groups): 

696 node.getparent().remove(node) 

697 fields.pop(node.get('name'), None) 

698 # no point processing view-level ``groups`` anymore, return 

699 return False 

700 if node.get('groups'): 

701 can_see = self.user_has_groups(groups=node.get('groups')) 

702 if not can_see: 

703 node.set('invisible', '1') 

704 modifiers['invisible'] = True 

705 if 'attrs' in node.attrib: 

706 del node.attrib['attrs'] # avoid making field visible later 

707 del node.attrib['groups'] 

708 return True

716 def postprocess(self, model, node, view_id, in_tree_view, model_fields): 

717 """Return the description of the fields in the node.

720 but it is actually possible to give some sub-node (this is used so 

721 that the method can call itself recursively). 

722 

723 Originally, the field descriptions are drawn from the node itself. 

724 But there is now some code calling fields_get() in order to merge some 

725 of those information in the architecture.

727 """ 

728 result = False 

729 fields = {} 

730 children = True 

731 

732 modifiers = {} 

733 if model not in self.env: 

734 self.raise_view_error(_('Model not found: %(model)s') % dict(model=model), view_id)

736 

737 if node.tag in ('field', 'node', 'arrow'): 

738 if node.get('object'): 

739 attrs = {}

744 'arch': xarch, 

745 'fields': xfields, 

746 } 

747 attrs = {'views': views} 

748 fields = xfields 

749 if node.get('name'): 

750 attrs = {} 

751 field = Model._fields.get(node.get('name')) 

752 if field: 

753 children = False 

754 views = {} 

755 for f in node: 

756 if f.tag in ('form', 'tree', 'graph', 'kanban', 'calendar'): 

757 node.remove(f) 

758 xarch, xfields = self.with_context(base_model_name=model).postprocess_and_fields(field.comodel_name, f, view_id)

760 'arch': xarch,

773 

774 elif node.tag in ('form', 'tree'):

783 fields[node.get(additional_field)] = {} 

784 

785 if not self._apply_group(model, node, modifiers, fields):

787 return fields

791 orm.transfer_node_to_modifiers(node, modifiers, self._context, in_tree_view) 

792 

793 for f in node: 

794 if children or (node.tag == 'field' and f.tag in ('filter', 'separator')):

796 

797 orm.transfer_modifiers_to_node(modifiers, node)

800 def add_on_change(self, model_name, arch): 

801 """ Add attribute on_change="1" on fields that are dependencies of

803 """

815 collect(child, model) 

816 

817 collect(arch, self.env[model_name]) 

818 

819 for field, nodes in field_nodes.iteritems():

822 model = self.env[field.model_name] 

823 if model._has_onchange(field, field_nodes): 

824 for node in nodes: 

825 if not node.get('on_change'): 

826 node.set('on_change', '1') 

827 

828 return arch

978 def distribute_branding(self, e, branding=None, parent_xpath='', 

979 index_map=ConstantMapping(1)): 

980 if e.get('t-ignore') or e.tag == 'head': 

981 # remove any view branding possibly injected by inheritance 

982 attrs = set(MOVABLE_BRANDING) 

983 for descendant in e.iterdescendants(tag=etree.Element): 

984 if not attrs.intersection(descendant.attrib): 

985 continue 

986 self._pop_view_branding(descendant) 

987 # TODO: find a better name and check if we have a string to boolean helper 

988 return 

989 

990 node_path = e.get('data-oe-xpath') 

991 if node_path is None: 

992 node_path = "%s/%s[%d]" % (parent_xpath, e.tag, index_map[e.tag]) 

993 if branding and not (e.get('data-oe-model') or e.get('t-field')): 

994 e.attrib.update(branding) 

995 e.set('data-oe-xpath', node_path) 

996 if not e.get('data-oe-model'): 

997 return 

998 

999 if {'t-esc', 't-raw'}.intersection(e.attrib): 

1000 # nodes which fully generate their content and have no reason to 

1001 # be branded because they can not sensibly be edited 

1002 self._pop_view_branding(e) 

1003 elif self._contains_branded(e): 

1004 # if a branded element contains branded elements distribute own 

1005 # branding to children unless it's t-raw, then just remove branding 

1006 # on current element 

1007 distributed_branding = self._pop_view_branding(e) 

1008 

1009 if 't-raw' not in e.attrib: 

1010 # TODO: collections.Counter if remove p2.6 compat 

1011 # running index by tag type, for XPath query generation 

1012 indexes = collections.defaultdict(lambda: 0) 

1013 for child in e.iterchildren(tag=etree.Element): 

1014 if child.get('data-oe-xpath'): 

1015 # injected by view inheritance, skip otherwise 

1016 # generated xpath is incorrect 

1017 self.distribute_branding(child) 

1018 else: 

1019 indexes[child.tag] += 1 

1020 self.distribute_branding( 

1021 child, distributed_branding,
