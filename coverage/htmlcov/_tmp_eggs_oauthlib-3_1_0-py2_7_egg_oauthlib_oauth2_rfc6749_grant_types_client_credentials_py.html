18class ClientCredentialsGrant(GrantTypeBase): 

19 

20 """`Client Credentials Grant`_ 

21 

22 The client can request an access token using only its client 

23 credentials (or other supported means of authentication) when the 

24 client is requesting access to the protected resources under its 

25 control, or those of another resource owner that have been previously 

26 arranged with the authorization server (the method of which is beyond 

27 the scope of this specification). 

28 

29 The client credentials grant type MUST only be used by confidential 

30 clients:: 

31 

32 +---------+ +---------------+ 

33 : : : : 

34 : :>-- A - Client Authentication --->: Authorization : 

35 : Client : : Server : 

36 : :<-- B ---- Access Token ---------<: : 

37 : : : : 

38 +---------+ +---------------+ 

39 

40 Figure 6: Client Credentials Flow 

41 

42 The flow illustrated in Figure 6 includes the following steps: 

43 

44 (A) The client authenticates with the authorization server and 

45 requests an access token from the token endpoint. 

46 

47 (B) The authorization server authenticates the client, and if valid, 

48 issues an access token. 

49 

50 .. _`Client Credentials Grant`: https://tools.ietf.org/html/rfc6749#section-4.4 

51 """ 

52 

53 def create_token_response(self, request, token_handler): 

54 """Return token or error in JSON format. 

55 

56 :param request: OAuthlib request. 

57 :type request: oauthlib.common.Request 

58 :param token_handler: A token handler instance, for example of type 

59 oauthlib.oauth2.BearerToken. 

60 

61 If the access token request is valid and authorized, the 

62 authorization server issues an access token as described in 

63 `Section 5.1`_. A refresh token SHOULD NOT be included. If the request 

64 failed client authentication or is invalid, the authorization server 

65 returns an error response as described in `Section 5.2`_. 

66 

67 .. _`Section 5.1`: https://tools.ietf.org/html/rfc6749#section-5.1 

68 .. _`Section 5.2`: https://tools.ietf.org/html/rfc6749#section-5.2 

69 """ 

70 headers = self._get_default_headers() 

71 try: 

72 log.debug('Validating access token request, %r.', request) 

73 self.validate_token_request(request) 

74 except errors.OAuth2Error as e: 

75 log.debug('Client error in token request. %s.', e) 

76 headers.update(e.headers) 

77 return headers, e.json, e.status_code 

78 

79 token = token_handler.create_token(request, refresh_token=False) 

80 

81 for modifier in self._token_modifiers: 

82 token = modifier(token) 

83 

84 self.request_validator.save_token(token, request) 

85 

86 log.debug('Issuing token to client id %r (%r), %r.', 

87 request.client_id, request.client, token) 

88 return headers, json.dumps(token), 200
