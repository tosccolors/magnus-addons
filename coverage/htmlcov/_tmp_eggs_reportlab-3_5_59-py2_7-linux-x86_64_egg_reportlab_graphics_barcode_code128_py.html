173class Code128(MultiWidthBarcode): 

174 """ 

175 Code 128 is a very compact symbology that can encode the entire 

176 128 character ASCII set, plus 4 special control codes, 

177 (FNC1-FNC4, expressed in the input string as \xf1 to \xf4). 

178 Code 128 can also encode digits at double density (2 per byte) 

179 and has a mandatory checksum. Code 128 is well supported and 

180 commonly used -- for example, by UPS for tracking labels. 

181  

182 Because of these qualities, Code 128 is probably the best choice 

183 for a linear symbology today (assuming you have a choice). 

184 

185 Options that may be passed to constructor: 

186 

187 value (int, or numeric string. required.): 

188 The value to encode. 

189  

190 barWidth (float, default .0075): 

191 X-Dimension, or width of the smallest element 

192 Minumum is .0075 inch (7.5 mils). 

193  

194 barHeight (float, see default below): 

195 Height of the symbol. Default is the height of the two 

196 bearer bars (if they exist) plus the greater of .25 inch 

197 or .15 times the symbol's length. 

198 

199 quiet (bool, default 1): 

200 Wether to include quiet zones in the symbol. 

201  

202 lquiet (float, see default below): 

203 Quiet zone size to left of code, if quiet is true. 

204 Default is the greater of .25 inch, or 10 barWidth 

205  

206 rquiet (float, defaults as above): 

207 Quiet zone size to right left of code, if quiet is true. 

208  

209 Sources of Information on Code 128: 

210 

211 http://www.semiconductor.agilent.com/barcode/sg/Misc/code_128.html 

212 http://www.adams1.com/pub/russadam/128code.html 

213 http://www.barcodeman.com/c128.html 

214 

215 Official Spec, "ANSI/AIM BC4-1999, ISS" is available for US$45 from 

216 http://www.aimglobal.org/aimstore/ 

217 """ 

218 barWidth = inch * 0.0075 

219 lquiet = None 

220 rquiet = None 

221 quiet = 1 

222 barHeight = None

239 def validate(self): 

240 vval = "" 

241 self.valid = 1 

242 for c in self.value: 

243 if ord(c) > 127 and c not in '\xf1\xf2\xf3\xf4': 

244 self.valid = 0 

245 continue 

246 vval = vval + c 

247 self.validated = vval 

248 return vval

251 def _try_TO_C(self, l): 

252 '''Improved version of old _trailingDigitsToC(self, l) inspired by''' 

253 i = 0 

254 nl = [] 

255 while i < len(l): 

256 startpos = i 

257 rl = [] 

258 savings = -1 # the TO_C costs one character 

259 while i < len(l): 

260 if l[i] in cStarts: 

261 j = i 

262 break 

263 elif l[i] == '\xf1': 

264 rl.append(l[i]) 

265 i += 1 

266 continue 

267 elif l[i] in digits \ 

268 and l[i+1] in digits: 

269 rl.append(l[i] + l[i+1]) 

270 i += 2 

271 savings += 1 

272 continue 

273 else: 

274 if l[i] in digits and l[i+1]=='STOP': 

275 rrl = [] 

276 rsavings = -1 #we need a TO_C 

277 k = i 

278 while k>startpos: 

279 if l[k]=='\xf1': 

280 rrl.append(l[i]) 

281 k -= 1 

282 elif l[k] in digits and l[k-1] in digits: 

283 rrl.append(l[k-1]+l[k]) 

284 rsavings += 1 

285 k -= 2 

286 else: 

287 break 

288 rrl.reverse() 

289 if rsavings>savings+int(savings>=0 and (startpos and nl[-1] in cStarts))-1: 

290 nl += l[startpos] 

291 startpos += 1 

292 rl = rrl 

293 del rrl 

294 i += 1 

295 break 

296 ta = not (l[i]=='STOP' or j==i) 

297 xs = savings>=0 and (startpos and nl[-1] in cStarts) 

298 if savings+int(xs) > int(ta): 

299 if xs: 

300 toc = nl[-1][:-1]+'C' 

301 del nl[-1] 

302 else: 

303 toc = 'TO_C' 

304 nl += [toc]+rl 

305 if ta: 

306 nl.append('TO'+l[j][-2:]) 

307 nl.append(l[i]) 

308 else: 

309 nl += l[startpos:i+1] 

310 i += 1 

311 return nl

313 def encode(self): 

314 # First, encode using only B 

315 s = self.validated 

316 l = ['START_B'] 

317 for c in s: 

318 if c not in setb: 

319 l = l + ['TO_A', c, 'TO_B'] 

320 else: 

321 l.append(c) 

322 l.append('STOP') 

323 

324 l = self._try_TO_C(l) 

325 

326 # Finally, replace START_X,TO_Y with START_Y 

327 if l[1] in tos: 

328 l[:2] = ['START_' + l[1][-1]] 

329 

330# print repr(l) 

331 

332 # encode into numbers 

333 start, set, shset = setmap[l[0]] 

334 e = [start] 

335 

336 l = l[1:-1] 

337 while l: 

338 c = l[0] 

339 if c == 'SHIFT': 

340 e = e + [set[c], shset[l[1]]] 

341 l = l[2:] 

342 elif c in tos: 

343 e.append(set[c]) 

344 set, shset = setmap[c] 

345 l = l[1:] 

346 else: 

347 e.append(set[c]) 

348 l = l[1:] 

349 

350 c = e[0] 

351 for i in range(1, len(e)): 

352 c = c + i * e[i] 

353 self.encoded = e + [c % 103, stop] 

354 return self.encoded

357 self.decomposed = ''.join([_patterns[c] for c in self.encoded]) 

358 return self.decomposed
