546def getArgvDict(**kw): 

547 ''' Builds a dictionary from its keyword arguments with overrides from sys.argv. 

548 Attempts to be smart about conversions, but the value can be an instance 

549 of ArgDictValue to allow specifying a conversion function. 

550 ''' 

551 def handleValue(v,av,func): 

552 if func: 

553 v = func(av) 

554 else: 

555 if isStr(v): 

556 v = av 

557 elif isinstance(v,float): 

558 v = float(av) 

559 elif isinstance(v,int): 

560 v = int(av) 

561 elif isinstance(v,list): 

562 v = list(literal_eval(av),{}) 

563 elif isinstance(v,tuple): 

564 v = tuple(literal_eval(av),{}) 

565 else: 

566 raise TypeError("Can't convert string %r to %s" % (av,type(v))) 

567 return v 

568 

569 A = sys.argv[1:] 

570 R = {} 

571 for k, v in kw.items(): 

572 if isinstance(v,ArgvDictValue): 

573 v, func = v.value, v.func 

574 else: 

575 func = None 

576 handled = 0 

577 ke = k+'=' 

578 for a in A: 

579 if a.startswith(ke): 

580 av = a[len(ke):] 

581 A.remove(a) 

582 R[k] = handleValue(v,av,func) 

583 handled = 1 

584 break 

585 

586 if not handled: R[k] = handleValue(v,v,func) 

587 

588 return R

1309def prev_this_next(items): 

1310 """ 

1311 Loop over a collection with look-ahead and look-back. 

1312  

1313 From Thomas Guest,  

1314 http://wordaligned.org/articles/zippy-triples-served-with-python 

1315  

1316 Seriously useful looping tool (Google "zippy triples") 

1317 lets you loop a collection and see the previous and next items, 

1318 which get set to None at the ends. 

1319  

1320 To be used in layout algorithms where one wants a peek at the 

1321 next item coming down the pipe. 

1322 

1323 """ 

1324 

1325 extend = itertools.chain([None], items, [None]) 

1326 prev, this, next = itertools.tee(extend, 3) 

1327 try: 

1328 next(this) 

1329 next(next) 

1330 next(next) 

1331 except StopIteration: 

1332 pass 

1333 return zip(prev, this, next)

1335def commasplit(s): 

1336 ''' 

1337 Splits the string s at every unescaped comma and returns the result as a list. 

1338 To escape a comma, double it. Individual items are stripped. 

1339 To avoid the ambiguity of 3 successive commas to denote a comma at the beginning 

1340 or end of an item, add a space between the item seperator and the escaped comma. 

1341  

1342 >>> commasplit(u'a,b,c') == [u'a', u'b', u'c'] 

1343 True 

1344 >>> commasplit('a,, , b , c ') == [u'a,', u'b', u'c'] 

1345 True 

1346 >>> commasplit(u'a, ,,b, c') == [u'a', u',b', u'c'] 

1347 ''' 

1348 if isBytes(s): s = s.decode('utf8') 

1349 n = len(s)-1 

1350 s += u' ' 

1351 i = 0 

1352 r=[u''] 

1353 while i<=n: 

1354 if s[i]==u',': 

1355 if s[i+1]==u',': 

1356 r[-1]+=u',' 

1357 i += 1 

1358 else: 

1359 r[-1] = r[-1].strip() 

1360 if i!=n: r.append(u'') 

1361 else: 

1362 r[-1] += s[i] 

1363 i+=1 

1364 r[-1] = r[-1].strip() 

1365 return r

1367def commajoin(l): 

1368 ''' 

1369 Inverse of commasplit, except that whitespace around items is not conserved. 

1370 Adds more whitespace than needed for simplicity and performance. 

1371  

1372 >>> commasplit(commajoin(['a', 'b', 'c'])) == [u'a', u'b', u'c'] 

1373 True 

1374 >>> commasplit((commajoin([u'a,', u' b ', u'c'])) == [u'a,', u'b', u'c'] 

1375 True 

1376 >>> commasplit((commajoin([u'a ', u',b', u'c'])) == [u'a', u',b', u'c']  

1377 ''' 

1378 return u','.join([ u' ' + asUnicode(i).replace(u',', u',,') + u' ' for i in l ])
