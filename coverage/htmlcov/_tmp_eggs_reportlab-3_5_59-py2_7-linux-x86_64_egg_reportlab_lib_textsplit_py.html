58def wordSplit(word, maxWidths, fontName, fontSize, encoding='utf8'): 

59 """Attempts to break a word which lacks spaces into two parts, the first of which 

60 fits in the remaining space. It is allowed to add hyphens or whatever it wishes. 

61 

62 This is intended as a wrapper for some language- and user-choice-specific splitting 

63 algorithms. It should only be called after line breaking on spaces, which covers western 

64 languages and is highly optimised already. It works on the 'last unsplit word'. 

65 

66 Presumably with further study one could write a Unicode splitting algorithm for text 

67 fragments whick was much faster. 

68 

69 Courier characters should be 6 points wide. 

70 >>> wordSplit('HelloWorld', 30, 'Courier', 10) 

71 [[0.0, 'Hello'], [0.0, 'World']] 

72 >>> wordSplit('HelloWorld', 31, 'Courier', 10) 

73 [[1.0, 'Hello'], [1.0, 'World']] 

74 """ 

75 if not isUnicode(word): 

76 uword = word.decode(encoding) 

77 else: 

78 uword = word 

79 

80 charWidths = getCharWidths(uword, fontName, fontSize) 

81 lines = dumbSplit(uword, charWidths, maxWidths) 

82 

83 if not isUnicode(word): 

84 lines2 = [] 

85 #convert back 

86 for (extraSpace, text) in lines: 

87 lines2.append([extraSpace, text.encode(encoding)]) 

88 lines = lines2 

89 

90 return lines

92def dumbSplit(word, widths, maxWidths): 

93 """This function attempts to fit as many characters as possible into the available 

94 space, cutting "like a knife" between characters. This would do for Chinese. 

95 It returns a list of (text, extraSpace) items where text is a Unicode string, 

96 and extraSpace is the points of unused space available on the line. This is a 

97 structure which is fairly easy to display, and supports 'backtracking' approaches 

98 after the fact. 

99 

100 Test cases assume each character is ten points wide... 

101 

102 >>> dumbSplit(u'Hello', [10]*5, 60) 

103 [[10, u'Hello']] 

104 >>> dumbSplit(u'Hello', [10]*5, 50) 

105 [[0, u'Hello']] 

106 >>> dumbSplit(u'Hello', [10]*5, 40) 

107 [[0, u'Hell'], [30, u'o']] 

108 """ 

109 _more = """ 

110 #>>> dumbSplit(u'Hello', [10]*5, 4) # less than one character 

111 #(u'', u'Hello') 

112 # this says 'Nihongo wa muzukashii desu ne!' (Japanese is difficult isn't it?) in 12 characters 

113 >>> jtext = u'\u65e5\u672c\u8a9e\u306f\u96e3\u3057\u3044\u3067\u3059\u306d\uff01' 

114 >>> dumbSplit(jtext, [10]*11, 30) # 

115 (u'\u65e5\u672c\u8a9e', u'\u306f\u96e3\u3057\u3044\u3067\u3059\u306d\uff01') 

116 """ 

117 if not isinstance(maxWidths,(list,tuple)): maxWidths = [maxWidths] 

118 assert isUnicode(word) 

119 lines = [] 

120 i = widthUsed = lineStartPos = 0 

121 maxWidth = maxWidths[0] 

122 nW = len(word) 

123 while i<nW: 

124 w = widths[i] 

125 c = word[i] 

126 widthUsed += w 

127 i += 1 

128 if widthUsed > maxWidth + _FUZZ and widthUsed>0: 

129 extraSpace = maxWidth - widthUsed 

130 if ord(c)<0x3000: 

131 # we appear to be inside a non-Asian script section. 

132 # (this is a very crude test but quick to compute). 

133 # This is likely to be quite rare so the speed of the 

134 # code below is hopefully not a big issue. The main 

135 # situation requiring this is that a document title 

136 # with an english product name in it got cut. 

137 

138 

139 # we count back and look for  

140 # - a space-like character 

141 # - reversion to Kanji (which would be a good split point) 

142 # - in the worst case, roughly half way back along the line 

143 limitCheck = (lineStartPos+i)>>1 #(arbitrary taste issue) 

144 for j in range(i-1,limitCheck,-1): 

145 cj = word[j] 

146 if category(cj)=='Zs' or ord(cj)>=0x3000: 

147 k = j+1 

148 if k<i: 

149 j = k+1 

150 extraSpace += sum(widths[j:i]) 

151 w = widths[k] 

152 c = word[k] 

153 i = j 

154 break 

155 

156 #end of English-within-Asian special case 

157 

158 #we are pushing this character back, but 

159 #the most important of the Japanese typography rules 

160 #if this character cannot start a line, wrap it up to this line so it hangs 

161 #in the right margin. We won't do two or more though - that's unlikely and 

162 #would result in growing ugliness. 

163 #and increase the extra space 

164 #bug fix contributed by Alexander Vasilenko <alexs.vasilenko@gmail.com> 

165 if c not in ALL_CANNOT_START and i>lineStartPos+1: 

166 #otherwise we need to push the character back 

167 #the i>lineStart+1 condition ensures progress 

168 i -= 1 

169 extraSpace += w 

170 

171 #lines.append([maxWidth-sum(widths[lineStartPos:i]), word[lineStartPos:i].strip()]) 

172 lines.append([extraSpace, word[lineStartPos:i].strip()]) 

173 try: 

174 maxWidth = maxWidths[len(lines)] 

175 except IndexError: 

176 maxWidth = maxWidths[-1] # use the last one 

177 lineStartPos = i 

178 widthUsed = 0 

179 

180 #any characters left? 

181 if widthUsed > 0: 

182 lines.append([maxWidth - widthUsed, word[lineStartPos:]]) 

183 

184 return lines
