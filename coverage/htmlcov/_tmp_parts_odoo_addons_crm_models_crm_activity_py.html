7class CrmActivity(models.Model): 

8 ''' CrmActivity is a model introduced in Odoo v9 that models activities 

9 performed in CRM, like phone calls, sending emails, making demonstrations, 

10 ... Users are able to configure their custom activities. 

11 

12 Each activity can configure recommended next activities. This allows to model 

13 light custom workflows. This way sales manager can configure their crm 

14 workflow that salepersons will use in their daily job. 

15 

16 CrmActivity inherits from mail.message.subtype. This allows users to follow 

17 some activities through subtypes. Each activity will generate messages with 

18 the matching subtypes, allowing reporting and statistics computation based 

19 on mail.message.subtype model. ''' 

20 

21 _name = 'crm.activity' 

22 _description = 'CRM Activity' 

23 _inherits = {'mail.message.subtype': 'subtype_id'} 

24 _rec_name = 'name' 

25 _order = "sequence" 

26 

27 days = fields.Integer('Number of days', default=0, 

28 help='Number of days before executing the action, allowing you to plan the date of the action.') 

29 sequence = fields.Integer('Sequence', default=0) 

30 team_id = fields.Many2one('crm.team', string='Sales Team') 

31 subtype_id = fields.Many2one('mail.message.subtype', string='Message Subtype', required=True, ondelete='cascade') 

32 recommended_activity_ids = fields.Many2many( 

33 'crm.activity', 'crm_activity_rel', 'activity_id', 'recommended_id', 

34 string='Recommended Next Activities') 

35 preceding_activity_ids = fields.Many2many( 

36 'crm.activity', 'crm_activity_rel', 'recommended_id', 'activity_id', 

37 string='Preceding Activities') 

38 

39 # setting a default value on inherited fields is a bit involved 

40 res_model = fields.Char('Model', related='subtype_id.res_model', inherited=True, default='crm.lead') 

41 internal = fields.Boolean('Internal Only', related='subtype_id.internal', inherited=True, default=True) 

42 default = fields.Boolean('Default', related='subtype_id.default', inherited=True, default=False)

44 @api.multi 

45 def unlink(self): 

46 for activity in self: 

47 activities = self.search([('subtype_id', '=', activity.subtype_id.id)]) 

48 # to ensure that the subtype is only linked the current activity 

49 if len(activities) == 1: 

50 activity.subtype_id.unlink() 

51 return super(CrmActivity, self).unlink()
