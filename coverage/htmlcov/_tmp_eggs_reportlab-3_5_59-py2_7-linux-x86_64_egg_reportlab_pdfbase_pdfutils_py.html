16def _chunker(src,dst=[],chunkSize=60): 

17 for i in range(0,len(src),chunkSize): 

18 dst.append(src[i:i+chunkSize]) 

19 return dst

29def makeA85Image(filename,IMG=None, detectJpeg=False): 

30 import zlib 

31 img = ImageReader(filename) 

32 if IMG is not None: 

33 IMG.append(img) 

34 if detectJpeg and img.jpeg_fh(): 

35 return None 

36 

37 imgwidth, imgheight = img.getSize() 

38 raw = img.getRGBData() 

39 

40 code = [] 

41 append = code.append 

42 # this describes what is in the image itself 

43 append('BI') 

44 append('/W %s /H %s /BPC 8 /CS /%s /F [/A85 /Fl]' % (imgwidth, imgheight,_mode2cs[img.mode])) 

45 append('ID') 

46 #use a flate filter and Ascii Base 85 

47 assert len(raw) == imgwidth * imgheight*_mode2bpp[img.mode], "Wrong amount of data for image" 

48 compressed = zlib.compress(raw) #this bit is very fast... 

49 encoded = asciiBase85Encode(compressed) #...sadly this may not be 

50 

51 #append in blocks of 60 characters 

52 _chunker(encoded,code) 

53 

54 append('EI') 

55 return code

83def cacheImageFile(filename, returnInMemory=0, IMG=None): 

84 "Processes image as if for encoding, saves to a file with .a85 extension." 

85 

86 cachedname = os.path.splitext(filename)[0] + (rl_config.useA85 and '.a85' or '.bin') 

87 if filename==cachedname: 

88 if cachedImageExists(filename): 

89 from reportlab.lib.utils import open_for_read 

90 if returnInMemory: return filter(None,open_for_read(cachedname).read().split('\r\n')) 

91 else: 

92 raise IOError('No such cached image %s' % filename) 

93 else: 

94 if rl_config.useA85: 

95 code = makeA85Image(filename,IMG) 

96 else: 

97 code = makeRawImage(filename,IMG) 

98 if returnInMemory: return code 

99 

100 #save it to a file 

101 f = open(cachedname,'wb') 

102 f.write('\r\n'.join(code)+'\r\n') 

103 f.close() 

104 if rl_config.verbose: 

105 print('cached image as %s' % cachedname)

108def preProcessImages(spec): 

109 """Preprocesses one or more image files. 

110 

111 Accepts either a filespec ('C:\\mydir\\*.jpg') or a list 

112 of image filenames, crunches them all to save time. Run this 

113 to save huge amounts of time when repeatedly building image 

114 documents.""" 

115 

116 import glob 

117 

118 if isinstance(spec,str): 

119 filelist = glob.glob(spec) 

120 else: #list or tuple OK 

121 filelist = spec 

122 

123 for filename in filelist: 

124 if cachedImageExists(filename): 

125 if rl_config.verbose: 

126 print('cached version of %s already exists' % filename) 

127 else: 

128 cacheImageFile(filename)

131def cachedImageExists(filename): 

132 """Determines if a cached image already exists for a given file. 

133 

134 Determines if a cached image exists which has the same name 

135 and equal or newer date to the given file.""" 

136 cachedname = os.path.splitext(filename)[0] + (rl_config.useA85 and '.a85' or 'bin') 

137 if os.path.isfile(cachedname): 

138 #see if it is newer 

139 original_date = os.stat(filename)[8] 

140 cached_date = os.stat(cachedname)[8] 

141 if original_date > cached_date: 

142 return 0 

143 else: 

144 return 1 

145 else: 

146 return 0

167def _AsciiHexEncode(input): 

168 """Encodes input using ASCII-Hex coding. 

169 

170 This is a verbose encoding used for binary data within 

171 a PDF file. One byte binary becomes two bytes of ASCII. 

172 Helper function used by images.""" 

173 if isUnicode(input): 

174 input = input.encode('utf-8') 

175 output = getBytesIO() 

176 output.write(binascii.b2a_hex(input)) 

177 output.write(b'>') 

178 return output.getvalue()

181def _AsciiHexDecode(input): 

182 """Decodes input using ASCII-Hex coding. 

183 

184 Not used except to provide a test of the inverse function.""" 

185 

186 #strip out all whitespace 

187 if not isUnicode(input): 

188 input = input.decode('utf-8') 

189 stripped = ''.join(input.split()) 

190 assert stripped[-1] == '>', 'Invalid terminator for Ascii Hex Stream' 

191 stripped = stripped[:-1] #chop off terminator 

192 assert len(stripped) % 2 == 0, 'Ascii Hex stream has odd number of bytes' 

193 

194 return ''.join([chr(int(stripped[i:i+2],16)) for i in range(0,len(stripped),2)])

196def _wrap(input, columns=60): 

197 "Wraps input at a given column size by inserting \r\n characters." 

198 output = [] 

199 length = len(input) 

200 i = 0 

201 pos = columns * i 

202 while pos < length: 

203 output.append(input[pos:pos+columns]) 

204 i = i + 1 

205 pos = columns * i 

206 #avoid HP printer problem 

207 if len(output[-1])==1: 

208 output[-2:] = [output[-2][:-1],output[-2][-1]+output[-1]] 

209 return '\r\n'.join(output)

218# Read data from the JPEG file. We should probably be using PIL to

220# Returns (width, height, color components) as a triple 

221# This is based on Thomas Merz's code from GhostScript (viewjpeg.ps) 

222def readJPEGInfo(image): 

223 "Read width, height and number of components from open JPEG file." 

224 

225 import struct 

226 from reportlab.pdfbase.pdfdoc import PDFError 

227 

228 #Acceptable JPEG Markers: 

229 # SROF0=baseline, SOF1=extended sequential or SOF2=progressive 

230 validMarkers = [0xC0, 0xC1, 0xC2] 

231 

232 #JPEG markers without additional parameters 

233 noParamMarkers = \ 

234 [ 0xD0, 0xD1, 0xD2, 0xD3, 0xD4, 0xD5, 0xD6, 0xD7, 0xD8, 0x01 ] 

235 

236 #Unsupported JPEG Markers 

237 unsupportedMarkers = \ 

238 [ 0xC3, 0xC5, 0xC6, 0xC7, 0xC8, 0xC9, 0xCA, 0xCB, 0xCD, 0xCE, 0xCF ] 

239 

240 #read JPEG marker segments until we find SOFn marker or EOF 

241 done = 0 

242 while not done: 

243 x = struct.unpack('B', image.read(1)) 

244 if x[0] == 0xFF: #found marker 

245 x = struct.unpack('B', image.read(1)) 

246 #print "Marker: ", '%0.2x' % x[0] 

247 #check marker type is acceptable and process it 

248 if x[0] in validMarkers: 

249 image.seek(2, 1) #skip segment length 

250 x = struct.unpack('B', image.read(1)) #data precision 

251 if x[0] != 8: 

252 raise PDFError('JPEG must have 8 bits per component') 

253 y = struct.unpack('BB', image.read(2)) 

254 height = (y[0] << 8) + y[1] 

255 y = struct.unpack('BB', image.read(2)) 

256 width = (y[0] << 8) + y[1] 

257 y = struct.unpack('B', image.read(1)) 

258 color = y[0] 

259 return width, height, color 

260 elif x[0] in unsupportedMarkers: 

261 raise PDFError('JPEG Unsupported JPEG marker: %0.2x' % x[0]) 

262 elif x[0] not in noParamMarkers: 

263 #skip segments with parameters 

264 #read length and skip the data 

265 x = struct.unpack('BB', image.read(2)) 

266 image.seek( (x[0] << 8) + x[1] - 2, 1)

268class _fusc: 

269 def __init__(self,k, n): 

270 assert k, 'Argument k should be a non empty string' 

271 self._k = k 

272 self._klen = len(k) 

273 self._n = int(n) or 7

275 def encrypt(self,s): 

276 return self.__rotate(asciiBase85Encode(''.join(map(chr,self.__fusc(list(map(ord,s)))))),self._n) 

277 

278 def decrypt(self,s): 

279 return ''.join(map(chr,self.__fusc(list(map(ord,asciiBase85Decode(self.__rotate(s,-self._n)))))))

281 def __rotate(self,s,n): 

282 l = len(s) 

283 if n<0: n = l+n 

284 n %= l 

285 if not n: return s 

286 return s[-n:]+s[:l-n]
