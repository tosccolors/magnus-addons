107 @api.multi 

108 def signup_prepare(self, signup_type="signup", expiration=False): 

109 """ generate a new token for the partners with the given validity, if necessary 

110 :param expiration: the expiration datetime of the token (string, optional) 

111 """ 

112 for partner in self: 

113 if expiration or not partner.signup_valid: 

114 token = random_token() 

115 while self._signup_retrieve_partner(token): 

116 token = random_token() 

117 partner.write({'signup_token': token, 'signup_type': signup_type, 'signup_expiration': expiration}) 

118 return True

120 @api.model 

121 def _signup_retrieve_partner(self, token, check_validity=False, raise_exception=False): 

122 """ find the partner corresponding to a token, and possibly check its validity 

123 :param token: the token to resolve 

124 :param check_validity: if True, also check validity 

125 :param raise_exception: if True, raise exception instead of returning False 

126 :return: partner (browse record) or False (if raise_exception is False) 

127 """ 

128 partner = self.search([('signup_token', '=', token)], limit=1) 

129 if not partner: 

130 if raise_exception: 

131 raise SignupError("Signup token '%s' is not valid" % token) 

132 return False 

133 if check_validity and not partner.signup_valid: 

134 if raise_exception: 

135 raise SignupError("Signup token '%s' is no longer valid" % token) 

136 return False 

137 return partner
