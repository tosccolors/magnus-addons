25def prepare_headers(oauth_params, headers=None, realm=None): 

26 """**Prepare the Authorization header.** 

27 Per `section 3.5.1`_ of the spec. 

28 

29 Protocol parameters can be transmitted using the HTTP "Authorization" 

30 header field as defined by `RFC2617`_ with the auth-scheme name set to 

31 "OAuth" (case insensitive). 

32 

33 For example:: 

34 

35 Authorization: OAuth realm="Example", 

36 oauth_consumer_key="0685bd9184jfhq22", 

37 oauth_token="ad180jjd733klru7", 

38 oauth_signature_method="HMAC-SHA1", 

39 oauth_signature="wOJIO9A2W5mFwDgiDvZbTSMK%2FPY%3D", 

40 oauth_timestamp="137131200", 

41 oauth_nonce="4572616e48616d6d65724c61686176", 

42 oauth_version="1.0" 

43 

44 

45 .. _`section 3.5.1`: https://tools.ietf.org/html/rfc5849#section-3.5.1 

46 .. _`RFC2617`: https://tools.ietf.org/html/rfc2617 

47 """ 

48 headers = headers or {} 

49 

50 # Protocol parameters SHALL be included in the "Authorization" header 

51 # field as follows: 

52 authorization_header_parameters_parts = [] 

53 for oauth_parameter_name, value in oauth_params: 

54 # 1. Parameter names and values are encoded per Parameter Encoding 

55 # (`Section 3.6`_) 

56 # 

57 # .. _`Section 3.6`: https://tools.ietf.org/html/rfc5849#section-3.6 

58 escaped_name = utils.escape(oauth_parameter_name) 

59 escaped_value = utils.escape(value) 

60 

61 # 2. Each parameter's name is immediately followed by an "=" character 

62 # (ASCII code 61), a """ character (ASCII code 34), the parameter 

63 # value (MAY be empty), and another """ character (ASCII code 34). 

64 part = '{0}="{1}"'.format(escaped_name, escaped_value) 

65 

66 authorization_header_parameters_parts.append(part) 

67 

68 # 3. Parameters are separated by a "," character (ASCII code 44) and 

69 # OPTIONAL linear whitespace per `RFC2617`_. 

70 # 

71 # .. _`RFC2617`: https://tools.ietf.org/html/rfc2617 

72 authorization_header_parameters = ', '.join( 

73 authorization_header_parameters_parts) 

74 

75 # 4. The OPTIONAL "realm" parameter MAY be added and interpreted per 

76 # `RFC2617 section 1.2`_. 

77 # 

78 # .. _`RFC2617 section 1.2`: https://tools.ietf.org/html/rfc2617#section-1.2 

79 if realm: 

80 # NOTE: realm should *not* be escaped 

81 authorization_header_parameters = ('realm="%s", ' % realm + 

82 authorization_header_parameters) 

83 

84 # the auth-scheme name set to "OAuth" (case insensitive). 

85 authorization_header = 'OAuth %s' % authorization_header_parameters 

86 

87 # contribute the Authorization header to the given headers 

88 full_headers = {} 

89 full_headers.update(headers) 

90 full_headers['Authorization'] = authorization_header 

91 return full_headers

94def _append_params(oauth_params, params): 

95 """Append OAuth params to an existing set of parameters. 

96 

97 Both params and oauth_params is must be lists of 2-tuples. 

98 

99 Per `section 3.5.2`_ and `3.5.3`_ of the spec. 

100 

101 .. _`section 3.5.2`: https://tools.ietf.org/html/rfc5849#section-3.5.2 

102 .. _`3.5.3`: https://tools.ietf.org/html/rfc5849#section-3.5.3 

103 

104 """ 

105 merged = list(params) 

106 merged.extend(oauth_params) 

107 # The request URI / entity-body MAY include other request-specific 

108 # parameters, in which case, the protocol parameters SHOULD be appended 

109 # following the request-specific parameters, properly separated by an "&" 

110 # character (ASCII code 38) 

111 merged.sort(key=lambda i: i[0].startswith('oauth_')) 

112 return merged
