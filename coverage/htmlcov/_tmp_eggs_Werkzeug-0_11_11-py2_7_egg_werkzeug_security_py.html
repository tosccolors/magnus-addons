36def _find_hashlib_algorithms(): 

37 algos = getattr(hashlib, 'algorithms', None) 

38 if algos is None: 

39 algos = ('md5', 'sha1', 'sha224', 'sha256', 'sha384', 'sha512') 

40 rv = {} 

41 for algo in algos: 

42 func = getattr(hashlib, algo, None) 

43 if func is not None: 

44 rv[algo] = func 

45 return rv 

46_hash_funcs = _find_hashlib_algorithms()

49def pbkdf2_hex(data, salt, iterations=DEFAULT_PBKDF2_ITERATIONS, 

50 keylen=None, hashfunc=None): 

51 """Like :func:`pbkdf2_bin`, but returns a hex-encoded string. 

52 

53 .. versionadded:: 0.9 

54 

55 :param data: the data to derive. 

56 :param salt: the salt for the derivation. 

57 :param iterations: the number of iterations. 

58 :param keylen: the length of the resulting key. If not provided, 

59 the digest size will be used. 

60 :param hashfunc: the hash function to use. This can either be the 

61 string name of a known hash function, or a function 

62 from the hashlib module. Defaults to sha1. 

63 """

71def pbkdf2_bin(data, salt, iterations=DEFAULT_PBKDF2_ITERATIONS, 

72 keylen=None, hashfunc=None): 

73 """Returns a binary digest for the PBKDF2 hash algorithm of `data` 

74 with the given `salt`. It iterates `iterations` times and produces a 

75 key of `keylen` bytes. By default, SHA-1 is used as hash function; 

76 a different hashlib `hashfunc` can be provided. 

77 

78 .. versionadded:: 0.9 

79 

80 :param data: the data to derive. 

81 :param salt: the salt for the derivation. 

82 :param iterations: the number of iterations. 

83 :param keylen: the length of the resulting key. If not provided 

84 the digest size will be used. 

85 :param hashfunc: the hash function to use. This can either be the 

86 string name of a known hash function or a function 

87 from the hashlib module. Defaults to sha1. 

88 """ 

89 if isinstance(hashfunc, string_types): 

90 hashfunc = _hash_funcs[hashfunc] 

91 elif not hashfunc: 

92 hashfunc = hashlib.sha1 

93 data = to_bytes(data) 

94 salt = to_bytes(salt) 

95 

96 # If we're on Python with pbkdf2_hmac we can try to use it for 

97 # compatible digests. 

98 if _has_native_pbkdf2: 

99 _test_hash = hashfunc() 

100 if hasattr(_test_hash, 'name') and \ 

101 _test_hash.name in _hash_funcs: 

102 return hashlib.pbkdf2_hmac(_test_hash.name, 

103 data, salt, iterations, 

104 keylen) 

105 

106 mac = hmac.HMAC(data, None, hashfunc) 

107 if not keylen: 

108 keylen = mac.digest_size 

109 

110 def _pseudorandom(x, mac=mac): 

111 h = mac.copy() 

112 h.update(x) 

113 return bytearray(h.digest()) 

114 buf = bytearray() 

115 for block in range_type(1, -(-keylen // mac.digest_size) + 1): 

116 rv = u = _pseudorandom(salt + _pack_int(block)) 

117 for i in range_type(iterations - 1): 

118 u = _pseudorandom(bytes(u)) 

119 rv = bytearray(starmap(xor, izip(rv, u))) 

120 buf.extend(rv) 

121 return bytes(buf[:keylen])

124def safe_str_cmp(a, b): 

125 """This function compares strings in somewhat constant time. This 

126 requires that the length of at least one string is known in advance. 

127 

128 Returns `True` if the two strings are equal, or `False` if they are not. 

129 

130 .. versionadded:: 0.7 

131 """ 

132 if isinstance(a, text_type): 

133 a = a.encode('utf-8') 

134 if isinstance(b, text_type): 

135 b = b.encode('utf-8') 

136 

137 if _builtin_safe_str_cmp is not None: 

138 return _builtin_safe_str_cmp(a, b) 

139 

140 if len(a) != len(b): 

141 return False 

142 

143 rv = 0 

144 if PY2: 

145 for x, y in izip(a, b): 

146 rv |= ord(x) ^ ord(y) 

147 else: 

148 for x, y in izip(a, b): 

149 rv |= x ^ y 

150 

151 return rv == 0

158 return ''.join(_sys_rng.choice(SALT_CHARS) for _ in range_type(length))

161def _hash_internal(method, salt, password): 

162 """Internal password hash helper. Supports plaintext without salt, 

163 unsalted and salted passwords. In case salted passwords are used 

164 hmac is used. 

165 """ 

166 if method == 'plain': 

167 return password, method 

168 

169 if isinstance(password, text_type): 

170 password = password.encode('utf-8') 

171 

172 if method.startswith('pbkdf2:'): 

173 args = method[7:].split(':') 

174 if len(args) not in (1, 2): 

175 raise ValueError('Invalid number of arguments for PBKDF2') 

176 method = args.pop(0) 

177 iterations = args and int(args[0] or 0) or DEFAULT_PBKDF2_ITERATIONS 

178 is_pbkdf2 = True 

179 actual_method = 'pbkdf2:%s:%d' % (method, iterations) 

180 else: 

181 is_pbkdf2 = False 

182 actual_method = method 

183 

184 hash_func = _hash_funcs.get(method) 

185 if hash_func is None: 

186 raise TypeError('invalid method %r' % method) 

187 

188 if is_pbkdf2: 

189 if not salt: 

190 raise ValueError('Salt is required for PBKDF2') 

191 rv = pbkdf2_hex(password, salt, iterations, 

192 hashfunc=hash_func) 

193 elif salt: 

194 if isinstance(salt, text_type): 

195 salt = salt.encode('utf-8') 

196 rv = hmac.HMAC(salt, password, hash_func).hexdigest() 

197 else: 

198 h = hash_func() 

199 h.update(password) 

200 rv = h.hexdigest() 

201 return rv, actual_method

204def generate_password_hash(password, method='pbkdf2:sha1', salt_length=8): 

205 """Hash a password with the given method and salt with with a string of 

206 the given length. The format of the string returned includes the method 

207 that was used so that :func:`check_password_hash` can check the hash. 

208 

209 The format for the hashed string looks like this:: 

210 

211 method$salt$hash 

212 

213 This method can **not** generate unsalted passwords but it is possible 

214 to set the method to plain to enforce plaintext passwords. If a salt 

215 is used, hmac is used internally to salt the password. 

216 

217 If PBKDF2 is wanted it can be enabled by setting the method to 

218 ``pbkdf2:method:iterations`` where iterations is optional:: 

219 

220 pbkdf2:sha1:2000$salt$hash 

221 pbkdf2:sha1$salt$hash 

222 

223 :param password: the password to hash. 

224 :param method: the hash method to use (one that hashlib supports). Can 

225 optionally be in the format ``pbkdf2:<method>[:iterations]`` 

226 to enable PBKDF2. 

227 :param salt_length: the length of the salt in letters. 

228 """ 

229 salt = method != 'plain' and gen_salt(salt_length) or '' 

230 h, actual_method = _hash_internal(method, salt, password) 

231 return '%s$%s$%s' % (actual_method, salt, h)

234def check_password_hash(pwhash, password): 

235 """check a password against a given salted and hashed password value. 

236 In order to support unsalted legacy passwords this method supports 

237 plain text passwords, md5 and sha1 hashes (both salted and unsalted). 

238 

239 Returns `True` if the password matched, `False` otherwise. 

240 

241 :param pwhash: a hashed string like returned by 

242 :func:`generate_password_hash`. 

243 :param password: the plaintext password to compare against the hash. 

244 """ 

245 if pwhash.count('$') < 2: 

246 return False 

247 method, salt, hashval = pwhash.split('$', 2) 

248 return safe_str_cmp(_hash_internal(method, salt, password)[0], hashval)

251def safe_join(directory, filename): 

252 """Safely join `directory` and `filename`. If this cannot be done, 

253 this function returns ``None``. 

254 

255 :param directory: the base directory. 

256 :param filename: the untrusted filename relative to that directory. 

257 """ 

258 filename = posixpath.normpath(filename) 

259 for sep in _os_alt_seps: 

260 if sep in filename: 

261 return None 

262 if os.path.isabs(filename) or filename.startswith('../'): 

263 return None 

264 return os.path.join(directory, filename)
