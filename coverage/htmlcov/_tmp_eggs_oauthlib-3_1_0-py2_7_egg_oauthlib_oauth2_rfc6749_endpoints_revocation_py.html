41 @catch_errors_and_unavailability 

42 def create_revocation_response(self, uri, http_method='POST', body=None, 

43 headers=None): 

44 """Revoke supplied access or refresh token. 

45 

46 

47 The authorization server responds with HTTP status code 200 if the 

48 token has been revoked sucessfully or if the client submitted an 

49 invalid token. 

50 

51 Note: invalid tokens do not cause an error response since the client 

52 cannot handle such an error in a reasonable way. Moreover, the purpose 

53 of the revocation request, invalidating the particular token, is 

54 already achieved. 

55 

56 The content of the response body is ignored by the client as all 

57 necessary information is conveyed in the response code. 

58 

59 An invalid token type hint value is ignored by the authorization server 

60 and does not influence the revocation response. 

61 """ 

62 resp_headers = { 

63 'Content-Type': 'application/json', 

64 'Cache-Control': 'no-store', 

65 'Pragma': 'no-cache', 

66 } 

67 request = Request( 

68 uri, http_method=http_method, body=body, headers=headers) 

69 try: 

70 self.validate_revocation_request(request) 

71 log.debug('Token revocation valid for %r.', request) 

72 except OAuth2Error as e: 

73 log.debug('Client error during validation of %r. %r.', request, e) 

74 response_body = e.json 

75 if self.enable_jsonp and request.callback: 

76 response_body = '%s(%s);' % (request.callback, response_body) 

77 resp_headers.update(e.headers) 

78 return resp_headers, response_body, e.status_code 

79 

80 self.request_validator.revoke_token(request.token, 

81 request.token_type_hint, request) 

82 

83 response_body = '' 

84 if self.enable_jsonp and request.callback: 

85 response_body = request.callback + '();' 

86 return {}, response_body, 200

88 def validate_revocation_request(self, request): 

89 """Ensure the request is valid. 

90 

91 The client constructs the request by including the following parameters 

92 using the "application/x-www-form-urlencoded" format in the HTTP 

93 request entity-body: 

94 

95 token (REQUIRED). The token that the client wants to get revoked. 

96 

97 token_type_hint (OPTIONAL). A hint about the type of the token 

98 submitted for revocation. Clients MAY pass this parameter in order to 

99 help the authorization server to optimize the token lookup. If the 

100 server is unable to locate the token using the given hint, it MUST 

101 extend its search accross all of its supported token types. An 

102 authorization server MAY ignore this parameter, particularly if it is 

103 able to detect the token type automatically. This specification 

104 defines two such values: 

105 

106 * access_token: An Access Token as defined in [RFC6749], 

107 `section 1.4`_ 

108 

109 * refresh_token: A Refresh Token as defined in [RFC6749], 

110 `section 1.5`_ 

111 

112 Specific implementations, profiles, and extensions of this 

113 specification MAY define other values for this parameter using 

114 the registry defined in `Section 4.1.2`_. 

115 

116 The client also includes its authentication credentials as described in 

117 `Section 2.3`_. of [`RFC6749`_]. 

118 

119 .. _`section 1.4`: https://tools.ietf.org/html/rfc6749#section-1.4 

120 .. _`section 1.5`: https://tools.ietf.org/html/rfc6749#section-1.5 

121 .. _`section 2.3`: https://tools.ietf.org/html/rfc6749#section-2.3 

122 .. _`Section 4.1.2`: https://tools.ietf.org/html/draft-ietf-oauth-revocation-11#section-4.1.2 

123 .. _`RFC6749`: https://tools.ietf.org/html/rfc6749 

124 """
