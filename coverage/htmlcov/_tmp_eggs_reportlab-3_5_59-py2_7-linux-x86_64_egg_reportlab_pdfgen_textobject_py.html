69 def setFillColor(self, aColor, alpha=None): 

70 """Takes a color object, allowing colors to be referred to by name"""

81 self._code.append('%s k' % fp_str(c, m, y, k))

87 l = len(aColor) 

88 if l==3: 

89 self._fillColorObj = aColor

91 elif l==4: 

92 self._fillColorObj = aColor 

93 self._code.append('%s k' % fp_str(aColor))

174class PDFTextObject(_PDFColorSetter): 

175 """PDF logically separates text and graphics drawing; text 

176 operations need to be bracketed between BT (Begin text) and 

177 ET operators. This class ensures text operations are 

178 properly encapusalted. Ask the canvas for a text object 

179 with beginText(x, y). Do not construct one directly. 

180 Do not use multiple text objects in parallel; PDF is 

181 not multi-threaded! 

182 

183 It keeps track of x and y coordinates relative to its origin.""" 

184 

185 def __init__(self, canvas, x=0,y=0, direction=None): 

186 self._code = ['BT'] #no point in [] then append RGB 

187 self._canvas = canvas #canvas sets this so it has access to size info 

188 self._fontname = self._canvas._fontname 

189 self._fontsize = self._canvas._fontsize 

190 self._leading = self._canvas._leading 

191 self._doc = self._canvas._doc 

192 self._colorsUsed = self._canvas._colorsUsed 

193 self._enforceColorSpace = getattr(canvas,'_enforceColorSpace',None) 

194 font = pdfmetrics.getFont(self._fontname) 

195 self._curSubset = -1 

196 self.direction = direction 

197 self.setTextOrigin(x, y) 

198 self._textRenderMode = 0 

199 self._clipping = 0 

200 

201 def getCode(self): 

202 "pack onto one line; used internally" 

203 self._code.append('ET') 

204 if self._clipping: 

205 self._code.append('%d Tr' % (self._textRenderMode^4)) 

206 return ' '.join(self._code)

208 def setTextOrigin(self, x, y): 

209 if self._canvas.bottomup: 

210 self._code.append('1 0 0 1 %s Tm' % fp_str(x, y)) #bottom up 

211 else: 

212 self._code.append('1 0 0 -1 %s Tm' % fp_str(x, y)) #top down 

213 

214 # The current cursor position is at the text origin 

215 self._x0 = self._x = x 

216 self._y0 = self._y = y 

217 

218 def setTextTransform(self, a, b, c, d, e, f): 

219 "Like setTextOrigin, but does rotation, scaling etc." 

220 if not self._canvas.bottomup: 

221 c = -c #reverse bottom row of the 2D Transform 

222 d = -d 

223 self._code.append('%s Tm' % fp_str(a, b, c, d, e, f)) 

224 

225 # The current cursor position is at the text origin Note that 

226 # we aren't keeping track of all the transform on these 

227 # coordinates: they are relative to the rotations/sheers 

228 # defined in the matrix. 

229 self._x0 = self._x = e 

230 self._y0 = self._y = f

232 def moveCursor(self, dx, dy): 

233 """Starts a new line at an offset dx,dy from the start of the 

234 current line. This does not move the cursor relative to the 

235 current position, and it changes the current offset of every 

236 future line drawn (i.e. if you next do a textLine() call, it 

237 will move the cursor to a position one line lower than the 

238 position specificied in this call. """ 

239 

240 # Check if we have a previous move cursor call, and combine 

241 # them if possible. 

242 if self._code and self._code[-1][-3:]==' Td': 

243 L = self._code[-1].split() 

244 if len(L)==3: 

245 del self._code[-1] 

246 else: 

247 self._code[-1] = ''.join(L[:-4]) 

248 

249 # Work out the last movement 

250 lastDx = float(L[-3]) 

251 lastDy = float(L[-2]) 

252 

253 # Combine the two movement 

254 dx += lastDx 

255 dy -= lastDy 

256 

257 # We will soon add the movement to the line origin, so if 

258 # we've already done this for lastDx, lastDy, remove it 

259 # first (so it will be right when added back again). 

260 self._x0 -= lastDx 

261 self._y0 -= lastDy 

262 

263 # Output the move text cursor call. 

264 self._code.append('%s Td' % fp_str(dx, -dy)) 

265 

266 # Keep track of the new line offsets and the cursor position 

267 self._x0 += dx 

268 self._y0 += dy 

269 self._x = self._x0 

270 self._y = self._y0

349 def setTextRenderMode(self, mode): 

350 """Set the text rendering mode. 

351 

352 0 = Fill text 

353 1 = Stroke text 

354 2 = Fill then stroke 

355 3 = Invisible 

356 4 = Fill text and add to clipping path 

357 5 = Stroke text and add to clipping path 

358 6 = Fill then stroke and add to clipping path 

359 7 = Add to clipping path 

360 

361 after we start clipping we mustn't change the mode back until after the ET 

362 """ 

363 

364 assert mode in (0,1,2,3,4,5,6,7), "mode must be in (0,1,2,3,4,5,6,7)" 

365 if (mode & 4)!=self._clipping: 

366 mode |= 4 

367 self._clipping = mode & 4 

368 if self._textRenderMode!=mode: 

369 self._textRenderMode = mode 

370 self._code.append('%d Tr' % mode)

378 def _formatText(self, text): 

379 "Generates PDF text output operator(s)" 

380 if log2vis and self.direction in ('LTR','RTL'): 

381 # Use pyfribidi to write the text in the correct visual order. 

382 text = log2vis(text, directionsMap.get(self.direction,DIR_ON),clean=True) 

383 canv = self._canvas 

384 font = pdfmetrics.getFont(self._fontname) 

385 R = [] 

386 if font._dynamicFont: 

387 #it's a truetype font and should be utf8. If an error is raised, 

388 for subset, t in font.splitString(text, canv._doc): 

389 if subset!=self._curSubset: 

390 pdffontname = font.getSubsetInternalName(subset, canv._doc) 

391 R.append("%s %s Tf %s TL" % (pdffontname, fp_str(self._fontsize), fp_str(self._leading))) 

392 self._curSubset = subset 

393 R.append("(%s) Tj" % canv._escape(t)) 

394 elif font._multiByte: 

395 #all the fonts should really work like this - let them know more about PDF... 

396 R.append("%s %s Tf %s TL" % ( 

397 canv._doc.getInternalFontName(font.fontName), 

398 fp_str(self._fontsize), 

399 fp_str(self._leading) 

400 )) 

401 R.append("(%s) Tj" % font.formatForPdf(text)) 

402 else: 

403 #convert to T1 coding 

404 fc = font 

405 if isBytes(text): 

406 try: 

407 text = text.decode('utf8') 

408 except UnicodeDecodeError as e: 

409 i,j = e.args[2:4] 

410 raise UnicodeDecodeError(*(e.args[:4]+('%s\n%s-->%s<--%s' % (e.args[4],text[max(i-10,0):i],text[i:j],text[j:j+10]),))) 

411 

412 for f, t in pdfmetrics.unicode2T1(text,[font]+font.substitutionFonts): 

413 if f!=fc: 

414 R.append("%s %s Tf %s TL" % (canv._doc.getInternalFontName(f.fontName), fp_str(self._fontsize), fp_str(self._leading)))

416 R.append("(%s) Tj" % canv._escape(t)) 

417 if font!=fc: 

418 R.append("%s %s Tf %s TL" % (canv._doc.getInternalFontName(self._fontname), fp_str(self._fontsize), fp_str(self._leading)))

430 def textLine(self, text=''): 

431 """prints string at current point, text cursor moves down. 

432 Can work with no argument to simply move the cursor down.""" 

433 # Update the coordinates of the cursor 

434 self._x = self._x0 

435 if self._canvas.bottomup: 

436 self._y = self._y - self._leading 

437 else: 

438 self._y = self._y + self._leading 

439 

440 # Update the location of the start of the line 

441 # self._x0 is unchanged 

442 self._y0 = self._y 

443 

444 # Output the text followed by a PDF newline command 

445 self._code.append('%s T*' % self._formatText(text))

447 def textLines(self, stuff, trim=1): 

448 """prints multi-line or newlined strings, moving down. One 

449 comon use is to quote a multi-line block in your Python code; 

450 since this may be indented, by default it trims whitespace 

451 off each line and from the beginning; set trim=0 to preserve 

452 whitespace.""" 

453 if isStr(stuff): 

454 lines = asUnicode(stuff).strip().split(u'\n') 

455 if trim==1: 

456 lines = [s.strip() for s in lines] 

457 elif isinstance(stuff,(tuple,list)): 

458 lines = stuff 

459 else: 

460 assert 1==0, "argument to textlines must be string,, list or tuple" 

461 

462 # Output each line one at a time. This used to be a long-hand 

463 # copy of the textLine code, now called as a method. 

464 for line in lines: 

465 self.textLine(line)
