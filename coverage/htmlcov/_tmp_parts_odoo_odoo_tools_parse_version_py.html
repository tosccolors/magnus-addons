12def _parse_version_parts(s): 

13 for part in component_re.split(s): 

14 part = replace(part,part) 

15 if not part or part=='.': 

16 continue 

17 if part[:1] in '0123456789': 

18 yield part.zfill(8) # pad for numeric comparison 

19 else: 

20 yield '*'+part 

21 

22 yield '*final' # ensure that alpha/beta/candidate are before final

24def parse_version(s): 

25 """Convert a version string to a chronologically-sortable key 

26 

27 This is a rough cross between distutils' StrictVersion and LooseVersion; 

28 if you give it versions that would work with StrictVersion, then it behaves 

29 the same; otherwise it acts like a slightly-smarter LooseVersion. It is 

30 *possible* to create pathological version coding schemes that will fool 

31 this parser, but they should be very rare in practice. 

32 

33 The returned value will be a tuple of strings. Numeric portions of the 

34 version are padded to 8 digits so they will compare numerically, but 

35 without relying on how numbers compare relative to strings. Dots are 

36 dropped, but dashes are retained. Trailing zeros between alpha segments 

37 or dashes are suppressed, so that e.g. "2.4.0" is considered the same as 

38 "2.4". Alphanumeric parts are lower-cased. 

39 

40 The algorithm assumes that strings like "-" and any alpha string that 

41 alphabetically follows "final" represents a "patch level". So, "2.4-1" 

42 is assumed to be a branch or patch of "2.4", and therefore "2.4.1" is 

43 considered newer than "2.4-1", whic in turn is newer than "2.4". 

44 

45 Strings like "a", "b", "c", "alpha", "beta", "candidate" and so on (that 

46 come before "final" alphabetically) are assumed to be pre-release versions, 

47 so that the version "2.4" is considered newer than "2.4a1". 

48 

49 Finally, to handle miscellaneous cases, the strings "pre", "preview", and 

50 "rc" are treated as if they were "c", i.e. as though they were release 

51 candidates, and therefore are not as new as a version string that does not 

52 contain them. 

53 """ 

54 parts = [] 

55 for part in _parse_version_parts((s or '0.1').lower()): 

56 if part.startswith('*'): 

57 if part<'*final': # remove '-' before a prerelease tag 

58 while parts and parts[-1]=='*final-': parts.pop() 

59 # remove trailing zeros from each series of numeric parts 

60 while parts and parts[-1]=='00000000': 

61 parts.pop() 

62 parts.append(part) 

63 return tuple(parts)
