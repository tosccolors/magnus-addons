41def backup(path, raise_exception=True): 

42 path = os.path.normpath(path) 

43 if not os.path.exists(path): 

44 if not raise_exception: 

45 return None 

46 raise OSError('path does not exists') 

47 cnt = 1 

48 while True: 

49 bck = '%s~%d' % (path, cnt) 

50 if not os.path.exists(bck): 

51 shutil.move(path, bck) 

52 return bck 

53 cnt += 1

321 @api.multi 

322 def state_update(self, newstate, states_to_update, level=100): 

323 if level < 1: 

324 raise UserError(_('Recursion error in modules dependencies !')) 

325 

326 # whether some modules are installed with demo data 

327 demo = False 

328 

329 for module in self: 

330 # determine dependency modules to update/others 

331 update_mods, ready_mods = self.browse(), self.browse() 

332 for dep in module.dependencies_id: 

333 if dep.state == 'unknown': 

334 raise UserError(_("You try to install module '%s' that depends on module '%s'.\nBut the latter module is not available in your system.") % (module.name, dep.name,)) 

335 if dep.depend_id.state == newstate: 

336 ready_mods += dep.depend_id 

337 else: 

338 update_mods += dep.depend_id 

339 

340 # update dependency modules that require it, and determine demo for module 

341 update_demo = update_mods.state_update(newstate, states_to_update, level=level-1) 

342 module_demo = module.demo or update_demo or any(mod.demo for mod in ready_mods) 

343 demo = demo or module_demo 

344 

345 # check dependencies and update module itself 

346 self.check_external_dependencies(module.name, newstate) 

347 if module.state in states_to_update: 

348 module.write({'state': newstate, 'demo': module_demo}) 

349 

350 return demo

352 @api.multi 

353 def button_install(self): 

354 # domain to select auto-installable (but not yet installed) modules 

355 auto_domain = [('state', '=', 'uninstalled'), ('auto_install', '=', True)] 

356 

357 # determine whether an auto-install module must be installed: 

358 # - all its dependencies are installed or to be installed, 

359 # - at least one dependency is 'to install' 

360 install_states = frozenset(('installed', 'to install', 'to upgrade')) 

361 def must_install(module): 

362 states = set(dep.state for dep in module.dependencies_id) 

363 return states <= install_states and 'to install' in states 

364 

365 modules = self 

366 while modules: 

367 # Mark the given modules and their dependencies to be installed. 

368 modules.state_update('to install', ['uninstalled']) 

369 

370 # Determine which auto-installable modules must be installed. 

371 modules = self.search(auto_domain).filtered(must_install) 

372 

373 # retrieve the installed (or to be installed) theme modules 

374 theme_category = self.env.ref('base.module_category_theme') 

375 theme_modules = self.search([ 

376 ('state', 'in', list(install_states)), 

377 ('category_id', 'child_of', [theme_category.id]), 

378 ]) 

379 

380 # determine all theme modules that mods depends on, including mods 

381 def theme_deps(mods): 

382 deps = mods.mapped('dependencies_id.depend_id') 

383 while deps: 

384 mods |= deps 

385 deps = deps.mapped('dependencies_id.depend_id') 

386 return mods & theme_modules 

387 

388 if any(module.state == 'to install' for module in theme_modules): 

389 # check: the installation is valid if all installed theme modules 

390 # correspond to one theme module and all its theme dependencies 

391 if not any(theme_deps(module) == theme_modules for module in theme_modules): 

392 state_labels = dict(self.fields_get(['state'])['state']['selection']) 

393 themes_list = [ 

394 "- %s (%s)" % (module.shortdesc, state_labels[module.state]) 

395 for module in theme_modules 

396 ] 

397 raise UserError(_( 

398 "You are trying to install incompatible themes:\n%s\n\n" \ 

399 "Please uninstall your current theme before installing another one.\n" 

400 "Warning: switching themes may significantly alter the look of your current website pages!" 

401 ) % ("\n".join(themes_list)))

600 res = [0, 0] # [update, add] 

601 

602 default_version = modules.adapt_version('1.0') 

603 known_mods = self.with_context(lang=None).search([]) 

604 known_mods_names = {mod.name: mod for mod in known_mods} 

605 

606 # iterate through detected modules and update/create them in db 

607 for mod_name in modules.get_modules(): 

608 mod = known_mods_names.get(mod_name) 

609 terp = self.get_module_info(mod_name) 

610 values = self.get_values_from_terp(terp) 

611 

612 if mod: 

613 updated_values = {} 

614 for key in values: 

615 old = getattr(mod, key) 

616 updated = tools.ustr(values[key]) if isinstance(values[key], basestring) else values[key] 

617 if (old or updated) and updated != old: 

618 updated_values[key] = values[key] 

619 if terp.get('installable', True) and mod.state == 'uninstallable': 

620 updated_values['state'] = 'uninstalled' 

621 if parse_version(terp.get('version', default_version)) > parse_version(mod.latest_version or default_version): 

622 res[0] += 1 

623 if updated_values: 

624 mod.write(updated_values) 

625 else: 

626 mod_path = modules.get_module_path(mod_name) 

627 if not mod_path or not terp: 

628 continue 

629 state = "uninstalled" if terp.get('installable', True) else "uninstallable" 

630 mod = self.create(dict(name=mod_name, state=state, **values)) 

631 res[1] += 1 

632 

633 mod._update_dependencies(terp.get('depends', [])) 

634 mod._update_category(terp.get('category', 'Uncategorized')) 

635 

636 return res

642 @api.model 

643 def install_from_urls(self, urls): 

644 if not self.env.user.has_group('base.group_system'): 

645 raise AccessDenied() 

646 

647 # One-click install is opt-in - cfr Issue #15225 

648 ad_dir = tools.config.addons_data_dir 

649 if not os.access(ad_dir, os.W_OK): 

650 msg = (_("Automatic install of downloaded Apps is currently disabled.") + "\n\n" + 

651 _("To enable it, make sure this directory exists and is writable on the server:") + 

652 "\n%s" % ad_dir) 

653 _logger.warning(msg) 

654 raise UserError(msg) 

655 

656 apps_server = urlparse.urlparse(self.get_apps_server()) 

657 

658 OPENERP = odoo.release.product_name.lower() 

659 tmp = tempfile.mkdtemp() 

660 _logger.debug('Install from url: %r', urls) 

661 try: 

662 # 1. Download & unzip missing modules 

663 for module_name, url in urls.iteritems(): 

664 if not url: 

665 continue # nothing to download, local version is already the last one 

666 

667 up = urlparse.urlparse(url) 

668 if up.scheme != apps_server.scheme or up.netloc != apps_server.netloc: 

669 raise AccessDenied() 

670 

671 try: 

672 _logger.info('Downloading module `%s` from OpenERP Apps', module_name) 

673 content = urllib2.urlopen(url).read() 

674 except Exception: 

675 _logger.exception('Failed to fetch module %s', module_name) 

676 raise UserError(_('The `%s` module appears to be unavailable at the moment, please try again later.') % module_name) 

677 else: 

678 zipfile.ZipFile(StringIO(content)).extractall(tmp) 

679 assert os.path.isdir(os.path.join(tmp, module_name)) 

680 

681 # 2a. Copy/Replace module source in addons path 

682 for module_name, url in urls.iteritems(): 

683 if module_name == OPENERP or not url: 

684 continue # OPENERP is special case, handled below, and no URL means local module 

685 module_path = modules.get_module_path(module_name, downloaded=True, display_warning=False) 

686 bck = backup(module_path, False) 

687 _logger.info('Copy downloaded module `%s` to `%s`', module_name, module_path) 

688 shutil.move(os.path.join(tmp, module_name), module_path) 

689 if bck: 

690 shutil.rmtree(bck) 

691 

692 # 2b. Copy/Replace server+base module source if downloaded 

693 if urls.get(OPENERP): 

694 # special case. it contains the server and the base module. 

695 # extract path is not the same 

696 base_path = os.path.dirname(modules.get_module_path('base')) 

697 

698 # copy all modules in the SERVER/odoo/addons directory to the new "odoo" module (except base itself) 

699 for d in os.listdir(base_path): 

700 if d != 'base' and os.path.isdir(os.path.join(base_path, d)): 

701 destdir = os.path.join(tmp, OPENERP, 'addons', d) # XXX 'odoo' subdirectory ? 

702 shutil.copytree(os.path.join(base_path, d), destdir) 

703 

704 # then replace the server by the new "base" module 

705 server_dir = tools.config['root_path'] # XXX or dirname() 

706 bck = backup(server_dir) 

707 _logger.info('Copy downloaded module `odoo` to `%s`', server_dir) 

708 shutil.move(os.path.join(tmp, OPENERP), server_dir) 

709 #if bck: 

710 # shutil.rmtree(bck) 

711 

712 self.update_list() 

713 

714 with_urls = [module_name for module_name, url in urls.iteritems() if url] 

715 downloaded = self.search([('name', 'in', with_urls)]) 

716 installed = self.search([('id', 'in', downloaded.ids), ('state', '=', 'installed')]) 

717 

718 to_install = self.search([('name', 'in', urls.keys()), ('state', '=', 'uninstalled')]) 

719 post_install_action = to_install.button_immediate_install() 

720 

721 if installed or to_install: 

722 # in this case, force server restart to reload python code... 

723 self._cr.commit() 

724 odoo.service.server.restart() 

725 return { 

726 'type': 'ir.actions.client', 

727 'tag': 'home', 

728 'params': {'wait': True}, 

729 } 

730 return post_install_action 

731 

732 finally: 

733 shutil.rmtree(tmp)
