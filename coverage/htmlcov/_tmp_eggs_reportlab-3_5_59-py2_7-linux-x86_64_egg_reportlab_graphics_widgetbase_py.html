19 def verify(self): 

20 """If the _attrMap attribute is not None, this

52 def getProperties(self,recur=1): 

53 """Returns a list of all properties which can be edited and 

54 which are not marked as private. This may include 'child 

55 widgets' or 'primitive shapes'. You are free to override 

56 this and provide alternative implementations; the default 

57 one simply returns everything without a leading underscore.

65 props = {} 

66 for name in self.__dict__.keys(): 

67 if name[0:1] != '_': 

68 component = getattr(self, name) 

69 

70 if recur and isValidChild(component): 

71 # child object, get its properties too 

72 childProps = component.getProperties(recur=recur) 

73 for childKey, childValue in childProps.items(): 

74 #key might be something indexed like '[2].fillColor' 

75 #or simple like 'fillColor'; in the former case we

87 def setProperties(self, propDict): 

88 """Permits bulk setting of properties. These may include 

89 child objects e.g. "chart.legend.width = 200". 

90 

91 All assignments will be validated by the object as if they 

92 were set individually in python code. 

93 

94 All properties of a top-level object are guaranteed to be 

95 set before any of the children, which may be helpful to 

96 widget designers. 

97 """ 

98 

99 childPropDicts = {} 

100 for name, value in propDict.items(): 

101 parts = name.split('.', 1) 

102 if len(parts) == 1: 

103 #simple attribute, set it now 

104 setattr(self, name, value) 

105 else: 

106 (childName, remains) = parts 

107 try: 

108 childPropDicts[childName][remains] = value 

109 except KeyError: 

110 childPropDicts[childName] = {remains: value} 

111 

112 # now assign to children 

113 for childName, childPropDict in childPropDicts.items(): 

114 child = getattr(self, childName) 

115 child.setProperties(childPropDict)

187class TypedPropertyCollection(PropHolder): 

188 """A container with properties for objects of the same kind. 

189 

190 This makes it easy to create lists of objects. You initialize 

191 it with a class of what it is to contain, and that is all you 

192 can add to it. You can assign properties to the collection 

193 as a whole, or to a numeric index within it; if so it creates 

194 a new child object to hold that data. 

195 

196 So: 

197 wedges = TypedPropertyCollection(WedgeProperties) 

198 wedges.strokeWidth = 2 # applies to all 

199 wedges.strokeColor = colors.red # applies to all 

200 wedges[3].strokeColor = colors.blue # only to one 

201 

202 The last line should be taken as a prescription of how to 

203 create wedge no. 3 if one is needed; no error is raised if 

204 there are only two data points. 

205 

206 We try and make sensible use of tuple indices. 

207 line[(3,x)] is backed by line[(3,)] == line[3] & line 

208 """
