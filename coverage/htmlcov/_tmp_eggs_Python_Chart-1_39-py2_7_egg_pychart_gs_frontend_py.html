24def _get_gs_path(): 

25 """Guess where the Ghostscript executable is 

26 and return its absolute path name.""" 

27 path = os.environ.get("PATH", os.defpath) 

28 

29 for dir in path.split(os.pathsep): 

30 for name in ("gs", "gs.exe", "gswin32c.exe"): 

31 g = os.path.join(dir, name) 

32 if os.path.exists(g): 

33 return g 

34 raise Exception, "Ghostscript not found. path=%s" % str(path)

36class T(pscanvas.T): 

37 """This class is a special kind of canvas that runs ghostscript 

38 on the generated postscript contents. It is not used stand-alone, but as 

39 a component of PNG and X11 display functionality."""

41 def __write_contents(self, fp): 

42 fp.write(pscanvas.preamble_text) 

43 for name, id in self.__font_ids.items(): 

44 fp.write("/%s {/%s findfont SF} def\n" % (id, name)) 

45 fp.write("%d %d translate\n" % (-self.bbox[0], -self.bbox[1])) 

46 fp.writelines(self.__output_lines) 

47 fp.write("showpage end\n") 

48 fp.flush() 

49 

50 def close(self): 

51 # Don't call pscanvas.T.close, as it creates a 

52 # ps file.  

53 basecanvas.T.close(self)

55 def start_gs(self, arg): 

56 self.bbox = theme.adjust_bounding_box([xscale(self.__xmin), 

57 yscale(self.__ymin), 

58 xscale(self.__xmax), 

59 yscale(self.__ymax)]) 

60 

61 gs_path = _get_gs_path() 

62 self.pipe_fp = None 

63 if self.__output_lines == []: 

64 return 

65 

66 if sys.platform != "win32" and hasattr(os, "popen"): 

67 # UNIX-like systems 

68 cmdline = "\"%s\" -q %s -g%dx%d -q >/dev/null 2>&1" % \ 

69 (gs_path, arg, 

70 self.bbox[2] - self.bbox[0], 

71 self.bbox[3] - self.bbox[1]) 

72 self.pipe_fp = os.popen(cmdline, "w") 

73 self.__write_contents(self.pipe_fp) 

74 else: 

75 # XXX should use mktemp, but need to support python<=2.2 as well. 

76 fname = tempfile.mktemp("xxx") 

77 fp = open(fname, "wb") 

78 self.__write_contents(fp) 

79 fp.close() 

80 cmdline = "\"%s\" -q %s -g%dx%d -q <%s >NUL" % \ 

81 (gs_path, arg, 

82 self.bbox[2] - self.bbox[0], 

83 self.bbox[3] - self.bbox[1], fname) 

84 os.system(cmdline) 

85 os.unlink(fname)
