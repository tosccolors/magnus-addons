85class AnalyticLineBatchImporter(DelayedBatchImporter): 

86 """ Import the Jira worklogs 

87 

88 For every id in in the list, a delayed job is created. 

89 Import from a date 

90 """ 

91 _model_name = 'jira.account.analytic.line'

93 def run(self, from_date=None, to_date=None): 

94 """ Run the synchronization """ 

95 parts = [] 

96 if from_date: 

97 from_date = from_date.strftime(JIRA_JQL_DATETIME_FORMAT) 

98 parts.append('updated >= "%s"' % from_date) 

99 if to_date: 

100 to_date = to_date.strftime(JIRA_JQL_DATETIME_FORMAT) 

101 parts.append('updated <= "%s"' % to_date) 

102 issue_adapter = self.unit_for(JiraAdapter, model='jira.project.task') 

103 issue_ids = issue_adapter.search(' and '.join(parts)) 

104 for issue_id in issue_ids: 

105 for worklog_id in self.backend_adapter.search(issue_id): 

106 self._import_record(worklog_id, issue_id)

126 def _recurse_import_task(self): 

127 """ Import and return the task of proper type for the worklog 

128 

129 As we decide which type of issues are imported for a project, 

130 a worklog could be linked to an issue that we don't import. 

131 In that case, we climb the parents of the issue until we find 

132 a issue of a type we synchronize. 

133 

134 It ensures that the 'to-be-linked' issue is imported and return it. 

135 

136 """ 

137 issue_adapter = self.unit_for(JiraAdapter, model='jira.project.task') 

138 project_binder = self.binder_for('jira.project.project') 

139 issue_binder = self.binder_for('jira.project.task') 

140 issue_type_binder = self.binder_for('jira.issue.type') 

141 jira_issue_id = self.external_record['issueId'] 

142 epic_field_name = self.backend_record.epic_link_field_name 

143 current_project_id = self.external_issue['fields']['project']['id'] 

144 while jira_issue_id: 

145 issue = issue_adapter.read( 

146 jira_issue_id, 

147 fields=['issuetype', 'project', 'parent', epic_field_name], 

148 ) 

149 jira_project_id = issue['fields']['project']['id'] 

150 jira_issue_type_id = issue['fields']['issuetype']['id'] 

151 project_binding = project_binder.to_internal(jira_project_id) 

152 issue_type_binding = issue_type_binder.to_internal( 

153 jira_issue_type_id 

154 ) 

155 # JIRA allows to set an EPIC of a different project. 

156 # If it happens, we discard it. 

157 if (jira_project_id == current_project_id and 

158 issue_type_binding.is_sync_for_project(project_binding)): 

159 break 

160 if issue['fields'].get('parent'): 

161 # 'parent' is used on sub-tasks relating to their parent task 

162 jira_issue_id = issue['fields']['parent']['id'] 

163 elif issue['fields'].get(epic_field_name): 

164 # the epic link is set on a jira custom field 

165 epic_key = issue['fields'][epic_field_name] 

166 epic = issue_adapter.read(epic_key, fields='id') 

167 # we got the key of the epic issue, so we translate 

168 # it to the ID with a call to the API 

169 jira_issue_id = epic['id'] 

170 else: 

171 # no parent issue of a type we are synchronizing has been 

172 # found, the worklog will be assigned to no task 

173 jira_issue_id = None 

174 

175 if jira_issue_id: 

176 self._import_dependency(jira_issue_id, 'jira.project.task') 

177 return issue_binder.to_internal(jira_issue_id)
